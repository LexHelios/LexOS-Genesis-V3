FROM python:3.11-slim

# Install system dependencies
RUN apt-get update && apt-get install -y \
    curl \
    git \
    build-essential \
    libgl1 \
    libglib2.0-0 \
    && rm -rf /var/lib/apt/lists/*

# Add non-root user
RUN useradd -ms /bin/bash lexos
WORKDIR /app
USER lexos

# Copy requirements and install
COPY --chown=lexos:lexos requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy app
COPY --chown=lexos:lexos . .

RUN mkdir -p lexos_memory logs static temp config

EXPOSE 8080

ENV PYTHONUNBUFFERED=1
ENV OLLAMA_HOST=http://ollama:11434

# Healthcheck
HEALTHCHECK --interval=30s --timeout=10s --retries=3 \
  CMD curl -f http://localhost:8080/api/stats || exit 1

# Use uvicorn or python main.py for entrypoint
CMD ["python", "main.py"]
